<script>
 
    // initializing contextual menu
    var menu = new dhtmlXMenuObject("contextArea");
    menu.renderAsContextMenu();
    ...
    // adding onBeforeContextMenu event handler
    menu.attachEvent("onBeforeContextMenu", function(zoneId){
        // returning true allows contextual menu to be shown
        return true;
    });
    // adding onAfterContextMenu event handler
    menu.attachEvent("onAfterContextMenu", function(zoneId){
        // contextual menu was shown
    });
    // adding onClick event handler
    menu.attachEvent("onClick", function(id, zoneId){
        // contextual menu at zoneId element was clicked
    });
    // adding onRadioClick event handler
    menu.attachEvent("onRadioClick", function(group, idChecked, idClicked, zoneId){
        // contextual menu at zoneId radio button was clicked
    });
    // adding onCheckboxClick event handler
    menu.attachEvent("onCheckboxClick", function(id, state, zoneId){
        // contextual menu at zoneId checkbox was clicked
    });
</script>
